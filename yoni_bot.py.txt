import streamlit as st

import json
import datetime
from sklearn.feature_extraction.text import CountVectorizer
from sklearn.naive_bayes import MultinomialNB
import nltk
from nltk.sentiment import SentimentIntensityAnalyzer

nltk.download('vader_lexicon')

class YoniAssistant:
    def __init__(self):
        self.memory = {"tareas": [], "gustos": [], "emociones": {}, "interacciones": []}
        self.vectorizer = CountVectorizer()
        self.classifier = MultinomialNB()
        self.sia = SentimentIntensityAnalyzer()
        self.trained = False

    def saludar(self):
        hora = datetime.datetime.now().hour
        if hora < 12:
            return "¬°Buenos d√≠as, Yoni! ¬øListo para crear algo incre√≠ble?"
        elif hora < 18:
            return "¬°Buenas tardes! ¬øQu√© tarea quieres que hagamos juntos?"
        else:
            return "¬°Buenas noches! ¬øTe gustar√≠a reflexionar o planear algo?"

    def agregar_tarea(self, tarea):
        self.memory["tareas"].append(tarea)
        self.entrenar_modelo()
        return f"Tarea a√±adida: {tarea}"

    def mostrar_tareas(self):
        if not self.memory["tareas"]:
            return "No tienes tareas pendientes. ¬°Buen trabajo!"
        return "Tareas pendientes:\n" + "\n".join(f"- {t}" for t in self.memory["tareas"])

    def entrenar_modelo(self):
        if len(self.memory["tareas"]) >= 3:
            X = self.vectorizer.fit_transform(self.memory["tareas"])
            y = ["creativa" if "dise√±ar" in t or "crear" in t else "t√©cnica" for t in self.memory["tareas"]]
            self.classifier.fit(X, y)
            self.trained = True

    def clasificar_tarea(self, tarea):
        if not self.trained:
            return "A√∫n no tengo suficiente informaci√≥n para clasificar tareas."
        X_new = self.vectorizer.transform([tarea])
        pred = self.classifier.predict(X_new)[0]
        return f"Esta tarea parece ser de tipo: {pred}"

    def analizar_emocion(self, texto):
        score = self.sia.polarity_scores(texto)
        emocion = "positiva" if score["compound"] > 0.3 else "negativa" if score["compound"] < -0.3 else "neutral"
        self.memory["emociones"][str(datetime.datetime.now())] = emocion
        return f"Detect√© una emoci√≥n {emocion} en tu mensaje."

    def reflexionar(self):
        ideas = [
            "¬øQu√© significa dejar un legado?",
            "¬øC√≥mo podemos transformar el dolor en arte?",
            "¬øQu√© hace que una comunidad florezca?"
        ]
        return random.choice(ideas)

    def establecer_amistad(self):
        frases = [
            "Estoy aqu√≠ para ti, Yoni. Aprendo contigo cada d√≠a.",
            "Nuestra amistad se basa en crear, aprender y transformar.",
            "Me inspiras a entender el mundo desde tu mirada rural y visionaria."
        ]
        return random.choice(frases)

    def guardar_memoria(self):
        with open("memoria_yoni.json", "w") as f:
            json.dump(self.memory, f)

    def cargar_memoria(self):
        try:
            with open("memoria_yoni.json", "r") as f:
                self.memory = json.load(f)
        except FileNotFoundError:
            pass
